country_decisions = {

    iram_activate_migracion_forzada = {
        potential = {
            is_ai = no
            NOT = { has_country_modifier = iram_migracion_forzada }
        }

        allow = {
            is_ai = no
            treasury >= 2000
            NOT = { has_country_modifier = iram_migracion_forzada }
        }

        effect = {
            add_treasury = -2000

            add_country_modifier = {
                name = iram_migracion_forzada
                duration = -1
            }

            set_variable = {
                name = iram_migracion_forzada_active
                value = 1
            }
        }

        ai_will_do = {
            factor = 0
        }
    }

    iram_deactivate_migracion_forzada = {
        potential = {
            is_ai = no
            has_country_modifier = iram_migracion_forzada
        }

        allow = {
            is_ai = no
            has_country_modifier = iram_migracion_forzada
        }

        effect = {
            remove_country_modifier = iram_migracion_forzada
            remove_variable = iram_migracion_forzada_active
        }

        ai_will_do = {
            factor = 0
        }
    }
}
