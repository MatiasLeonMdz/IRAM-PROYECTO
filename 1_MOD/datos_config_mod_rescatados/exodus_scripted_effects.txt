exodus_cleanup_effect = {

    # Variables de operacion
    remove_variable = exodus_operation_active
    remove_variable = exodus_gather_active
    remove_variable = exodus_transfer_active
    remove_variable = exodus_distribute_active
    remove_variable = exodus_transfer_pending
    remove_variable = exodus_gather_pending
    remove_variable = exodus_distribute_pending
    remove_variable = exodus_anchor_province
    remove_variable = exodus_destination_province

    # Limpiar variables de province
    every_owned_province = {
        limit = {
            OR = {
                has_variable = exodus_is_anchor
                has_variable = exodus_is_destination
            }
        }
        remove_variable = exodus_is_anchor
        remove_variable = exodus_is_destination
    }

    # Disolver unidades Exodus
    every_unit = {
        limit = {
            OR = {
                has_variable = exodus_unit_transfer_origin
                has_variable = exodus_unit_transfer_dest
                has_variable = exodus_unit_concentrate
                has_variable = exodus_unit_distribute
            }
        }
        destroy_unit = yes
    }
}
